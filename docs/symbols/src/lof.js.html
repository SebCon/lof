 70   var queueManager = function() {
 71     var loop = function() {
 72       if (fx.length > 0) {
 73         var i = 0;
 74         setQueueHandling(true);
 75         //while (fx.length > 0) {
 76           var current = Date.now();
 77           if (current >= fx[i].timeStamp) {
 78             iterator(elements, fx[i].callback);
 79             fx.splice(i,1);
 80           } else {
 81             i++;
 82             if (i >= fx.length) {
 83               i = 0;
 84             }
 85           }
 86           if (fx.length > 0) {
 87             window.requestAnimationFrame(loop);
 88           } else {
 89             delay = 0;
 90           }
 91         }
 92     };
 93     window.requestAnimationFrame(loop);
 94     setQueueHandling(false);
 95   };

149   var getFactory = function(type, search) {
150     var elem = null;
151 
152     if (type) {
153       switch(type) {
154         case ELEMENT_ID:
155         elem = document.getElementById(search);
156         break;
157 
158         case ELEMENT_CLASS:
159         elem = getElemItems(document.getElementsByClassName(search));
160         break;
161 
162         case ELEMENT_TAG:
163         elem = getElemItems(document.getElementsByTagName(search));
164         break;
165 
166         case ELEMENT_NAME:
167         elem = getElemItems(document.getElementsByName(search));
168         break;
169 
170         case ELEMENT_CSS:
171         elem = getElemItems(document.querySelectorAll(search));
172         break;
173 
174         default:
175           console.warn('type ' + type + ' not found');
176       }
177     }
178 
179     return elem;
180 
181   };

196   /** get elements via id, class, tag, name or css
197   *		@function get
198   *		@param {Object} list config list
199   *   @param {string} list.id element id
200   *   @param {string} list.class element class
201   *   @param {string} list.tag element tag
202   *   @param {string} list.name element name
203   *   @param {string} list.css css selector
204   *
205   *   @return {this} lof object
206   **/
207   var get = function(list) {
208     if (list) {
209       var references = {};
210       references[ELEMENT_ID] = (list.id || null);
211       references[ELEMENT_CLASS] = (list.class || null);
212       references[ELEMENT_TAG] = (list.tag || null);
213       references[ELEMENT_NAME] = (list.name || null);
214       references[ELEMENT_CSS] = (list.css || null);
215       elements = [];
216 
217       for (var key in references) {
218         if (references[key] !== null) {
219           references[key] = transformArray(references[key]);
220           for (var k=0; k < references[key].length; k++) {
221             var elem = getFactory(key, references[key][k]);
222             if (elem) {
223               if (Array.isArray(elem)) {
224                 elements = elements.concat(elem);
225               } else {
226                 elements.push(elem);
227               }
228             }
229           }
230         }
231       }
232 
233     }
234 
235     return this;
236   };

435   var transformCSS = function(prop) {
436     var newProp = '';
437     var mGreat = false;
438     for (var i=0; i < prop.length; i++) {
439       if (prop[i] === '-') {
440         mGreat = true;
441       } else {
442           if (mGreat) {
443             mGreat = false;
444             newProp += prop[i].toUpperCase();
445           } else {
446             if (prop[i] !== ' ') {
447               newProp += prop[i].toLowerCase();
448             }
449           }
450       }
451     }
452 
453     return newProp;
454   };

457   /** get the value of the property | set the value of the property
458   *		@function css
459   *		@param {string} prop element property
460   *   @param {string} [value] property value
461   *
462   *   @return {string} value of property
463   **/
464   var css = function(prop, value) {
465     var back = [];
466     if (prop && value !== undefined && value !== null) {
467       prop = transformCSS(prop);
468       var callback = function(elem) {
469         if (elem) {
470           elem.style[prop] = value;
471         }
472       };
473       delayManager(callback);
474     } else if (prop) {
475       // props von den elementen ermitteln
476       var callback2 = function(elem) {
477         if (elem) {
478           back.push(elem.style[prop]);
479         }
480       };
481       iterator(elements, callback2);
482     }
483 
484     return back;
485   };
